# Clean column headers
meta_cols <- c("mz_rt", "Treatment", "Radiation", "Time")
expr_cols <- colnames(df)[!colnames(df) %in% meta_cols]
cleaned_expr_cols <- make.names(expr_cols, unique = TRUE)
colnames(df)[!colnames(df) %in% meta_cols] <- cleaned_expr_cols

# Impute expression columns using median values
df[, cleaned_expr_cols] <- lapply(df[, cleaned_expr_cols], function(x) as.numeric(as.character(x)))
df[, cleaned_expr_cols] <- lapply(df[, cleaned_expr_cols], function(x) {
  ifelse(is.na(x), median(x, na.rm = TRUE), x)
})

# Filter constant features to prevent calculation variance errors
non_const_cols <- sapply(df[, cleaned_expr_cols], function(x) {
  v <- var(x, na.rm = TRUE)
  !is.na(v) && v > 0
})
cleaned_expr_cols <- cleaned_expr_cols[non_const_cols]
df <- df[, c(meta_cols, cleaned_expr_cols)]

# Helper function to dynamically locate correct features
find_feature <- function(pattern, target_cols) {
  match <- grep(pattern, target_cols, ignore.case = TRUE, value = TRUE)
  if (length(match) > 0) return(match[1])
  stop(paste("Feature matching pattern", pattern, "not found!"))
}

# Resolve target marker names
f9_col   <- find_feature("F9", cleaned_expr_cols)
actb_col <- find_feature("ACTB", cleaned_expr_cols)
pfn1_col <- find_feature("PFN1", cleaned_expr_cols)
cfb_col  <- find_feature("CFB", cleaned_expr_cols)
ena_col  <- find_feature("ENA", cleaned_expr_cols)  # Represents 3,4-ENA
aa_col   <- find_feature("Arachid", cleaned_expr_cols) # Represents Arachidonic acid

# --- Figure 3C: Confusion Matrix ---
df_cm <- df_roc1[filtered_df$Radiation %in% c("PBI", "TBI"), ]
df_cm$Is_TBI <- ifelse(filtered_df$Radiation[filtered_df$Radiation %in% c("PBI", "TBI")] == "TBI", 1, 0)
model2 <- glm(Is_TBI ~ F9 + ACTB + PFN1 + CFB, data = df_cm, family = binomial)
df_cm$Pred <- predict(model2, type = "response")

roc2 <- pROC::roc(df_cm$Is_TBI, df_cm$Pred, quiet = TRUE)
cut_val <- as.numeric(pROC::coords(roc2, "best", ret = "threshold", best.method = "youden")[1])

df_cm$Pred_Cat <- factor(ifelse(df_cm$Pred >= cut_val, "TBI", "PBI"), levels = c("PBI", "TBI"))
df_cm$True_Cat <- factor(ifelse(df_cm$Is_TBI == 1, "TBI", "PBI"), levels = c("TBI", "PBI"))

  dplyr::group_by(True_Cat) %>% dplyr::mutate(Percent = round(n/sum(n)*100, 1))

fig3_cm <- ggplot(table_data, aes(x = Pred_Cat, y = True_Cat, fill = n)) +
  geom_tile(color = "black", linewidth = 1) +
  geom_text(aes(label = paste0(n, "\n(", Percent, "%)")), color = ifelse(table_data$n > max(table_data$n)/2, "white", "black"), size = 6, fontface = "bold") +
  scale_fill_gradient(low = "#F4F6F7", high = "#154360") +
  labs(title = "Figure 3C: PBI vs TBI Confusion Matrix", x = "Predicted Class", y = "True Exposure") +

cat("[6/6] Generating Figure 5 (Time-Course Trends)...\n")

# Normalize Time labels to correct order (Baseline to Day 6)
longitudinal_df <- df %>%
  dplyr::filter(Treatment == "VEH") %>%
  dplyr::mutate(Time = dplyr::case_when(
    Time == "SD-3" ~ "Baseline",
    Time %in% c("SD1", "1d") ~ "Day 1",
    Time %in% c("SD2", "2d") ~ "Day 2",
    Time %in% c("SD6", "6d") ~ "Day 6",
    TRUE ~ Time # 4H, 8H, 12H remain unchanged
  )) %>%
  dplyr::mutate(Time = factor(Time, levels = c("Baseline", "4H", "8H", "12H", "Day 1", "Day 2", "Day 6"))) %>%
  dplyr::filter(!is.na(Time))

# Calculate Mean and SEM for the 6 core biomarkers across PBI and TBI groups
summary_traj <- longitudinal_df %>%
  dplyr::select(Radiation, Time, dplyr::all_of(six_markers)) %>%
  tidyr::pivot_longer(cols = dplyr::all_of(six_markers), names_to = "Biomarker", values_to = "Value") %>%
  dplyr::group_by(Radiation, Time, Biomarker) %>%
  dplyr::summarise(
    Mean = mean(Value, na.rm = TRUE),
    SEM = sd(Value, na.rm = TRUE) / sqrt(dplyr::n()),
    .groups = "drop"
  )

# Label cleaner for graph titles
summary_traj <- summary_traj %>%
  dplyr::mutate(Clean_Label = dplyr::case_when(
    Biomarker == ena_col ~ "3,4-ENA",
    Biomarker == aa_col ~ "Arachidonic acid",
    TRUE ~ Biomarker
  ))

fig5_trajectories <- ggplot(summary_traj, aes(x = Time, y = Mean, group = Radiation, color = Radiation, fill = Radiation)) +
  geom_ribbon(aes(ymin = Mean - SEM, ymax = Mean + SEM), alpha = 0.15, color = NA) +
  geom_line(linewidth = 1.2) +
  geom_point(shape = 21, size = 2.5, stroke = 1, fill = "white") +
  facet_wrap(~ Clean_Label, scales = "free_y", nrow = 2) +
  scale_color_manual(values = c("PBI" = "#1E88E5", "TBI" = "#D32F2F")) +
  scale_fill_manual(values = c("PBI" = "#1E88E5", "TBI" = "#D32F2F")) +
  theme_bw(base_size = 12) +
